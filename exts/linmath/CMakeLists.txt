PROJECT(linmath)

SET(LINMATH_DIR ${CMAKE_CURRENT_SOURCE_DIR}/linmath)
SET(GLAD_DIR ${CMAKE_CURRENT_SOURCE_DIR}/glad/)

SET(LINMATH_SOURCE_GROUP_DIR ${LINMATH_DIR})
SET(GLAD_HEADER_GROUP_DIR ${GLAD_DIR}/include/glad ${GLAD_DIR}/include/KHR)
SET(GLAD_SOURCE_GROUP_DIR ${GLAD_DIR}/src)

FILE(GLOB LINMATH_FILES ${LINMATH_SOURCE_GROUP_DIR}/*.h)
FILE(GLOB GLAD_HEADER_FILES ${GLAD_HEADER_GROUP_DIR}/*.h)
FILE(GLOB GLAD_SOURCE_FILES ${GLAD_SOURCE_GROUP_DIR}/*.c)

SET(GLAD_FILES ${GLAD_HEADER_FILES} ${GLAD_SOURCE_FILES})

SOURCE_GROUP("exts/glad" FILES GLAD_HEADER_FILES GLAD_SOURCE_FILES)

set_target_properties (linmath PROPERTIES
    FOLDER C++
)
